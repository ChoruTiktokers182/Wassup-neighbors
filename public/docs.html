      async function login() {
        try {
          const response = await axios.post('link/login', {
            email: 'email/phone/fbid',
            pass: 'pass'
          });
          console.log('Login successful:', response.data);
        } catch (error) {
          console.error('Login failed:', error.response.data);
        }
      }

      login();
